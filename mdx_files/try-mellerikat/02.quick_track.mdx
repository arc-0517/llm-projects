export const QuickStartGuide = () => {
  // 활성화된 단계 상태 관리
  const [activeStep, setActiveStep] = useState('step1');

  // 각 단계별 참조 생성
  const stepRefs = {
    step1: useRef(null),
    step2: useRef(null),
    step3: useRef(null),
    step4: useRef(null),
    step5: useRef(null),
    step6: useRef(null),
    step7: useRef(null),
    step8: useRef(null),
  };

  // 단계 항목 클릭 핸들러
  const handleStepClick = (e, stepId) => {
    e.preventDefault();
    const element = stepRefs[stepId]?.current;
    if (element) {
      window.scrollTo({
        top: element.offsetTop - 100,
        behavior: 'smooth'
      });
      setActiveStep(stepId);
    }
  };

  // 코드 복사 핸들러
  const handleCopyCode = (codeText) => {
    navigator.clipboard.writeText(codeText).then(() => {
      // 성공적으로 복사되었을 때의 로직
    });
  };

  // 스크롤 감지 및 활성 단계 업데이트
  useEffect(() => {
    const handleScroll = () => {
      const scrollPosition = window.scrollY + 150;

      // 각 단계 요소의 위치 확인 및 활성 단계 설정
      Object.entries(stepRefs).forEach(([stepId, ref]) => {
        if (ref.current &&
            scrollPosition >= ref.current.offsetTop &&
            scrollPosition < (ref.current.offsetTop + ref.current.offsetHeight)) {
          setActiveStep(stepId);
        }
      });
    };

    window.addEventListener('scroll', handleScroll);
    return () => window.removeEventListener('scroll', handleScroll);
  }, []);

  return (
    <div className="quick-start-guide">
      <section className="hero">
        <div className="container">
          <div className="hero-content">
            <h1>Quick Start 트랙 상세 가이드</h1>
            <p>솔루션 개발과 엣지 설치 과정 없이 mellerikat의 핵심 기능을 빠르게 체험해보세요.</p>
          </div>
        </div>
      </section>

      <div className="container">
        {/* Overview Section */}
        <section className="overview">
